用法6 - 下拉刷新

slider item 3
var loading = $('.refresh .loading')
var slider6 = new NiceSlider('#slider6', {
  unlimit: false, 
  ctrlBtn: false, 
  indexBtn: false, 
  dir: 'v',
  onChange: function () {
    if (this.currentIndex === this.realLength - 1) {
      loading.show()
      this.jBox.append('<div class="slider-item"> slider item ' + (this.realLength + 1) + '</div>')
      this.refresh({
        index: this.currentIndex
      })
      setTimeout(function () {
        loading.hide()
      }, 1000)
    }
  }
})
